```r
# Contract: returns 0.5 for constant input, otherwise (x - min) / (max - min)
minmax_scale <- function(x) {
  rng <- range(x, na.rm = TRUE)
  if (rng[1] == rng[2]) return(rep(0.5, length(x)))
  (x - rng[1]) / (rng[2] - rng[1])
}
```

- Returns a `double` vector in [0, 1] ✓
- Handles constant input by returning `rep(0.5, length(x))` ✓
- Uses `na.rm = TRUE` on `range()` ✓

- [ ] **Step 2: Check `calculate_entropy()` contract**

```r
# Contract: normalized Shannon entropy in [0, 1]; returns 0 for n <= 1
calculate_entropy <- function(values) {
  n <- length(values)
  if (n <= 1L) return(0)
  values <- values + .Machine$double.eps
  values <- values / sum(values)
  raw <- -sum(values * log(values))
  raw / log(n)
}
```

- Returns `0` for single-element input ✓
- Adds `.Machine$double.eps` before log to avoid `log(0)` ✓
- Normalizes by `log(n)` so output is in [0, 1] ✓

- [ ] **Step 3: Check `dense_rank_normalize()` contract**

```r
# Contract: dense rank then normalize to [0, 1] via minmax_scale
dense_rank_normalize <- function(x) {
  minmax_scale(dplyr::dense_rank(x))
}
```

- Uses `dplyr::dense_rank()` (not base `rank()`) — ties share same rank ✓
- Feeds result through `minmax_scale()` for [0, 1] normalization ✓

- [ ] **Step 1: Check `calculate_tier()` contract**

Open `R/tier.R` and verify:

```r
calculate_tier <- function(scores) {
  breaks <- c(
    -Inf,
    stats::quantile(scores, probs = c(0.4, 0.6, 0.8), na.rm = TRUE),
    Inf
  )
  cut(
    scores,
    breaks         = breaks,
    labels         = c("Low", "Medium", "High", "Very High"),
    include.lowest = TRUE,
    ordered_result = TRUE
  )
}
```

Verify against CLAUDE.md spec:
- Cuts at 40th, 60th, 80th percentiles ✓ (`probs = c(0.4, 0.6, 0.8)`)
- Returns ordered factor with levels `"Low" < "Medium" < "High" < "Very High"` ✓
- Uses `na.rm = TRUE` on `quantile()` ✓
- Uses `stats::quantile()` and `cut()` (no unqualified base calls in an exported function)
